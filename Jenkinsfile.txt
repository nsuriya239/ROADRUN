     stages{
       /* stage('checkout'){

        stage('clean'){
		steps{
            echo 'Cleaning steps'
            withMaven(
                    // Maven installation declared in the Jenkins "Global Tool Configuration"
                    maven: 'MAVEN_HOME') {
                              // Run the maven build
                                bat 'mvn clean'
                   }
		 }
        }
         stage('test'){
		steps{
            echo 'Testing testcases'
            withMaven(
                    // Maven installation declared in the Jenkins "Global Tool Configuration"
                    maven: 'MAVEN_HOME') {
                              // Run the maven build
                                bat 'mvn clean test'
                   }
		 }
        }
        stage('results'){
            steps{
                echo 'Archiving results'
            archiveArtifacts artifacts: '**/test-output/*.*', fingerprint: true
            }
        }
     }
     post { 
        always { 
            echo 'This is Jenkins QA Automation Pipeline'
        }
    }

}
